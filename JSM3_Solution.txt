#include <stdio.h>
#include <string.h>
#include <assert.h>

void sortStrings(char arr[][50], int n) {
    char temp[50];
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (strcmp(arr[i], arr[j]) > 0) {
                strcpy(temp, arr[i]);
                strcpy(arr[i], arr[j]);
                strcpy(arr[j], temp);
            }
        }
    }
}

void testSortStrings() {
    char arr1[][50] = {"banana", "apple", "cherry"};
    sortStrings(arr1, 3);
    assert(strcmp(arr1[0], "apple") == 0);
    assert(strcmp(arr1[1], "banana") == 0);
    assert(strcmp(arr1[2], "cherry") == 0);

    char arr2[][50] = {"dog", "cat", "elephant", "bat"};
    sortStrings(arr2, 4);
    assert(strcmp(arr2[0], "bat") == 0);
    assert(strcmp(arr2[1], "cat") == 0);
    assert(strcmp(arr2[2], "dog") == 0);
    assert(strcmp(arr2[3], "elephant") == 0);
    
    char arr3[][50] = {"Banana", "apple", "cherry", "Orange"};
    sortStrings(arr3, 4);
    assert(strcmp(arr3[0], "Banana") == 0);
    assert(strcmp(arr3[1], "Orange") == 0);
    assert(strcmp(arr3[2], "apple") == 0);
    assert(strcmp(arr3[3], "cherry") == 0);

}

int main() {
    testSortStrings();
    printf("All string sorting tests passed.\n");
    return 0;
}
